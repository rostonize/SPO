package Test;

import java.util.List;
import java.util.ArrayList;


public class Test {
	public static void main (String args[]) {
		
		Lexer lexer = new Lexer("D:/EclipseExamples/Input.txt");

       
        System.out.println("-----------------");
        while (!lexer.isExhausted()) {
            System.out.printf("%-10s : %s \n", lexer.currentLexema(), lexer.currentToken());
            lexer.moveAhead();
        }
        System.out.println("-----------------");
        if (lexer.isSuccessful()) { //Если успешно дошли до конца файла, то выводим Done
            System.out.println("Done");
        } else {
            System.out.println(lexer.errorMessage());
        }
	}
}
